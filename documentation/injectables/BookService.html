    console.log(Globals.apiUrl);
    return this.http.get<any>(Globals.apiUrl + '/libros/all');

    console.log(`id ${id}`);
    return this.http.get<any>(Globals.apiUrl + '/libros/' + id);

    return this.http.get<any>(Globals.apiUrl + '/libros/autor/' + id);
